Ключи и списки.

const nums = [1, 2, 3, 4, 5];
const smartNums = [ { id: '1', value: 'smart 1' }, { id: '2', value: 'smart 2' }, { id: '3', value: 'smart 3' }, { id: '4', value: 'smart 4' }, { id: '5', value: 'smart 5' } ];

const ListItem = ({ value }) => (<li>{value}</li>);

const App = () => {
  const [array, setArray] = useState(nums);
  const [smartArray, setSmart] = useState(smartNums);

  const onClickArr = () => {
    const newArray = [...array];
    newArray.shift();
    setArray(newArray);
  };

  const onClickSmart = () => {
    const newArray = [...smartArray];
    newArray.shift();
    setSmart(newArray);
  };

  return (
    <>
      <button type='button' onClick={onClickArr}>Remove Arr</button>
      <ul>{array.map((num, index) => <ListItem key={index} value={num}/>)}</ul>
      <button type='button' onClick={onClickSmart}>Remove Smart</button>
      <ul>{smartArray.map(({id, value}) => <ListItem key={id} value={value}/>)}</ul>
    </>
)
}

Render Props

// ./components/Counter.js
const Counter = () => {
  const [count, setCount] = useState(0);
  const onClickPlus = () => setCount(count + 1);
  const onClickMinus = () => setCount(count - 1);

  return props.children({
    count,
    increment: onClickPlus,
    decrement: onClickMinus,
  });
};

// App.js
<RenderPropCounter>
  {({ count, increment, decrement }) => (
    <div className='d-inline-flex'>
      <Button onClick={decrement}>minus</Button>
      <Input value={count} disabled/>
      <Button onClick={increment}<plus>/Button>
    </div>
  )}
</RenderPropCounter>
