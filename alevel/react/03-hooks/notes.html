useState

function Example() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    document.title = `Счетчик: ${count}`;
  });

  return (
    <div>
      <p>Вы нажали {count} раз</p>
      <button onClick={() => setCount(count + 1)}>
        Нажми на меня
      </button>
    </div>
  );
}
